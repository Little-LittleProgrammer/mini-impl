# 渲染器详解

## 什么是渲染器？

渲染器负责将虚拟 DOM 转换为真实 DOM，并处理更新、卸载等操作。它是 Vue 运行时的核心模块。

```typescript
import { createRenderer, h } from 'vue'

const { render } = createRenderer({
  // 平台特定的操作
  createElement(tag) { return document.createElement(tag) },
  insert(child, parent, anchor) { parent.insertBefore(child, anchor) },
  // ...
})

// 渲染虚拟 DOM
render(h('div', null, 'Hello'), document.getElementById('app'))
```

## RendererOptions

渲染器通过 `RendererOptions` 接口抽象平台操作，实现平台无关性。

```typescript
// packages/runtime-core/src/renderer.ts

export interface RendererOptions {
  // 创建元素
  createElement(tag: string): Element

  // 创建文本节点
  createText(text: string): Text

  // 创建注释节点
  createComment(text: string): Comment

  // 设置文本内容
  setElementText(el: Element, text: string): void
  setText(node: Text, text: string): void

  // 插入节点
  insert(el: Node, parent: Element, anchor?: Node | null): void

  // 移除节点
  remove(el: Node): void

  // 查询父节点
  parentNode(node: Node): Element | null

  // 查询兄弟节点
  nextSibling(node: Node): Node | null

  // 属性更新
  patchProp(el: Element, key: string, prevValue: any, nextValue: any): void
}
```

## render 函数

`render()` 是渲染器的入口函数。

```typescript
const render: RootRenderFunction = (vnode, container) => {
  if (vnode == null) {
    // 卸载
    if (container._vnode) {
      unmount(container._vnode)
    }
  } else {
    // 挂载或更新
    patch(container._vnode || null, vnode, container)
  }
  // 缓存当前 vnode
  container._vnode = vnode
}
```

## patch 函数

`patch()` 是渲染的核心，负责对比新旧节点并执行相应操作。

```typescript
const patch = (
  n1: VNode | null,
  n2: VNode,
  container: Element,
  anchor: Node | null = null
) => {
  // 相同节点，跳过
  if (n1 === n2) return

  // 类型不同，卸载旧节点
  if (n1 && !isSameVNodeType(n1, n2)) {
    unmount(n1)
    n1 = null
  }

  const { type, shapeFlag } = n2

  switch (type) {
    case Text:
      processText(n1, n2, container, anchor)
      break

    case Comment:
      processComment(n1, n2, container, anchor)
      break

    case Fragment:
      processFragment(n1, n2, container, anchor)
      break

    default:
      if (shapeFlag & ShapeFlags.ELEMENT) {
        processElement(n1, n2, container, anchor)
      } else if (shapeFlag & ShapeFlags.COMPONENT) {
        processComponent(n1, n2, container, anchor)
      }
  }
}
```

## 处理元素节点

### processElement

```typescript
const processElement = (
  n1: VNode | null,
  n2: VNode,
  container: Element,
  anchor: Node | null
) => {
  if (n1 == null) {
    // 挂载
    mountElement(n2, container, anchor)
  } else {
    // 更新
    patchElement(n1, n2)
  }
}
```

### mountElement

```typescript
const mountElement = (
  vnode: VNode,
  container: Element,
  anchor: Node | null
) => {
  const { type, props, shapeFlag } = vnode

  // 1. 创建元素
  const el = (vnode.el = hostCreateElement(type as string))

  // 2. 处理属性
  if (props) {
    for (const key in props) {
      hostPatchProp(el, key, null, props[key])
    }
  }

  // 3. 处理子节点
  if (shapeFlag & ShapeFlags.TEXT_CHILDREN) {
    hostSetElementText(el, vnode.children as string)
  } else if (shapeFlag & ShapeFlags.ARRAY_CHILDREN) {
    mountChildren(vnode.children as VNode[], el, anchor)
  }

  // 4. 插入 DOM
  hostInsert(el, container, anchor)
}
```

### mountChildren

```typescript
const mountChildren = (
  children: VNode[],
  container: Element,
  anchor: Node | null
) => {
  for (let i = 0; i < children.length; i++) {
    const child = normalizeVNode(children[i])
    patch(null, child, container, anchor)
  }
}
```

### patchElement

```typescript
const patchElement = (
  n1: VNode,
  n2: VNode
) => {
  const el = (n2.el = n1.el!)
  const oldProps = n1.props || {}
  const newProps = n2.props || {}

  // 1. 更新属性
  patchProps(el, oldProps, newProps)

  // 2. 更新子节点
  const prevChildren = n1.children
  const nextChildren = n2.children

  if (n2.shapeFlag & ShapeFlags.TEXT_CHILDREN) {
    // 新子节点是文本
    if (prevChildren !== nextChildren) {
      hostSetElementText(el, nextChildren as string)
    }
  } else if (n2.shapeFlag & ShapeFlags.ARRAY_CHILDREN) {
    // 新子节点是数组
    if (n1.shapeFlag & ShapeFlags.ARRAY_CHILDREN) {
      // 都有子节点，执行 diff
      patchKeyedChildren(prevChildren as VNode[], nextChildren as VNode[], el)
    } else {
      // 旧节点没有子节点
      hostSetElementText(el, '')
      mountChildren(nextChildren as VNode[], el, null)
    }
  } else {
    // 新节点没有子节点
    if (n1.shapeFlag & ShapeFlags.ARRAY_CHILDREN) {
      unmountChildren(prevChildren as VNode[])
    } else if (prevChildren !== nextChildren) {
      hostSetElementText(el, '')
    }
  }
}
```

### patchProps

```typescript
const patchProps = (
  el: Element,
  oldProps: Record<string, any>,
  newProps: Record<string, any>
) => {
  // 更新和添加属性
  for (const key in newProps) {
    if (newProps[key] !== oldProps[key]) {
      hostPatchProp(el, key, oldProps[key], newProps[key])
    }
  }

  // 删除属性
  for (const key in oldProps) {
    if (!(key in newProps)) {
      hostPatchProp(el, key, oldProps[key], null)
    }
  }
}
```

## Diff 算法

Vue 3 使用高效的 diff 算法来比较新旧子节点列表。

### patchKeyedChildren

```typescript
const patchKeyedChildren = (
  c1: VNode[],
  c2: VNode[],
  container: Element,
  parentAnchor: Node | null
) => {
  let i = 0
  const l2 = c2.length
  let e1 = c1.length - 1
  let e2 = l2 - 1

  // 1. 从前往后比对相同前缀
  while (i <= e1 && i <= e2) {
    const n1 = c1[i]
    const n2 = c2[i]
    if (isSameVNodeType(n1, n2)) {
      patch(n1, n2, container)
    } else {
      break
    }
    i++
  }

  // 2. 从后往前比对相同后缀
  while (i <= e1 && i <= e2) {
    const n1 = c1[e1]
    const n2 = c2[e2]
    if (isSameVNodeType(n1, n2)) {
      patch(n1, n2, container)
    } else {
      break
    }
    e1--
    e2--
  }

  // 3. 新节点多于旧节点，挂载
  if (i > e1) {
    if (i <= e2) {
      const nextPos = e2 + 1
      const anchor = nextPos < l2 ? c2[nextPos].el : parentAnchor
      while (i <= e2) {
        patch(null, c2[i], container, anchor)
        i++
      }
    }
  }

  // 4. 旧节点多于新节点，卸载
  else if (i > e2) {
    while (i <= e1) {
      unmount(c1[i])
      i++
    }
  }

  // 5. 乱序处理
  else {
    const s1 = i
    const s2 = i
    const keyToNewIndexMap = new Map()

    // 5.1 建立 key -> newIndex 映射
    for (i = s2; i <= e2; i++) {
      const child = c2[i]
      if (child.key != null) {
        keyToNewIndexMap.set(child.key, i)
      }
    }

    let j
    let patched = 0
    const toBePatched = e2 - s2 + 1
    let moved = false
    let maxNewIndexSoFar = 0

    const newIndexToOldIndexMap = new Array(toBePatched)
    for (i = 0; i < toBePatched; i++) {
      newIndexToOldIndexMap[i] = 0
    }

    // 5.2 遍历旧节点，尝试匹配
    for (i = s1; i <= e1; i++) {
      const prevChild = c1[i]
      if (patched >= toBePatched) {
        // 所有新节点都已匹配，卸载剩余旧节点
        unmount(prevChild)
        continue
      }

      let newIndex
      if (prevChild.key != null) {
        newIndex = keyToNewIndexMap.get(prevChild.key)
      } else {
        // 无 key，遍历查找
        for (j = s2; j <= e2; j++) {
          if (newIndexToOldIndexMap[j - s2] === 0 && isSameVNodeType(prevChild, c2[j])) {
            newIndex = j
            break
          }
        }
      }

      if (newIndex === undefined) {
        // 没找到，卸载
        unmount(prevChild)
      } else {
        // 找到了，更新
        newIndexToOldIndexMap[newIndex - s2] = i + 1
        if (newIndex >= maxNewIndexSoFar) {
          maxNewIndexSoFar = newIndex
        } else {
          moved = true
        }
        patch(prevChild, c2[newIndex], container)
        patched++
      }
    }

    // 5.3 移动和挂载
    const increasingNewIndexSequence = moved
      ? getSequence(newIndexToOldIndexMap)
      : []

    j = increasingNewIndexSequence.length - 1
    for (i = toBePatched - 1; i >= 0; i--) {
      const nextIndex = s2 + i
      const nextChild = c2[nextIndex]
      const anchor = nextIndex + 1 < l2 ? c2[nextIndex + 1].el : parentAnchor

      if (newIndexToOldIndexMap[i] === 0) {
        // 新节点，挂载
        patch(null, nextChild, container, anchor)
      } else if (moved) {
        // 需要移动
        if (j < 0 || i !== increasingNewIndexSequence[j]) {
          move(nextChild, container, anchor)
        } else {
          j--
        }
      }
    }
  }
}
```

### Diff 算法流程

```
┌─────────────────────────────────────────────────────────────────┐
│                    Diff 算法流程                                 │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│  旧: [A, B, C, D, E, F]                                         │
│  新: [A, B, E, D, C, G]                                         │
│                                                                  │
│  1. 从前往后比对                                                 │
│     A == A, B == B                                              │
│     i = 2                                                       │
│                                                                  │
│  2. 从后往前比对                                                 │
│     无匹配                                                      │
│                                                                  │
│  3. 建立映射                                                     │
│     E -> 2, D -> 3, C -> 4, G -> 5                              │
│                                                                  │
│  4. 计算最长递增子序列                                           │
│     [D, C] -> [D]                                               │
│     increasingSequence = [3]                                    │
│                                                                  │
│  5. 移动和挂载                                                   │
│     - E: 移动到 D 前面                                          │
│     - D: 不动                                                   │
│     - C: 移动到 G 前面                                          │
│     - G: 新挂载                                                 │
│                                                                  │
└─────────────────────────────────────────────────────────────────┘
```

## 最长递增子序列

`getSequence` 函数计算最长递增子序列，用于最小化节点移动。

```typescript
function getSequence(arr: number[]): number[] {
  const p = arr.slice()
  const result = [0]
  let i, j, u, v, c
  const len = arr.length

  for (i = 0; i < len; i++) {
    const arrI = arr[i]
    if (arrI !== 0) {
      j = result[result.length - 1]
      if (arr[j] < arrI) {
        p[i] = j
        result.push(i)
        continue
      }

      u = 0
      v = result.length - 1
      while (u < v) {
        c = (u + v) >> 1
        if (arr[result[c]] < arrI) {
          u = c + 1
        } else {
          v = c
        }
      }

      if (arrI < arr[result[u]]) {
        if (u > 0) {
          p[i] = result[u - 1]
        }
        result[u] = i
      }
    }
  }

  u = result.length
  v = result[u - 1]
  while (u-- > 0) {
    result[u] = v
    v = p[v]
  }

  return result
}
```

## 卸载

```typescript
const unmount = (vnode: VNode) => {
  const { type, shapeFlag } = vnode

  if (type === Fragment) {
    // 卸载 Fragment 的所有子节点
    unmountChildren(vnode.children as VNode[])
  } else if (shapeFlag & ShapeFlags.COMPONENT) {
    // 卸载组件
    unmountComponent(vnode.component!)
  } else {
    // 移除 DOM 元素
    hostRemove(vnode.el!)
  }
}

const unmountChildren = (children: VNode[]) => {
  for (let i = 0; i < children.length; i++) {
    unmount(children[i])
  }
}
```

## 处理 Fragment

```typescript
const processFragment = (

  n2: VNode,

  if (n1 == null) {
```

## 处理文本节点

```typescript
const processText = (
  n1: VNode | null,
  n2: VNode,
  container: Element,
  anchor: Node | null
) => {
  if (n1 == null) {
    // 创建文本节点
    const el = (n2.el = hostCreateText(n2.children as string))
    hostInsert(el, container, anchor)
  } else {
    const el = (n2.el = n1.el!)
    if (n1.children !== n2.children) {
      hostSetText(el, n2.children as string)
    }
  }
}
```

## 总结

渲染器的核心功能：

| 功能 | 说明 |
|------|------|
| 平台抽象 | 通过 RendererOptions 接口 |
| patch | 对比新旧节点，执行相应操作 |
| diff 算法 | 高效比较子节点列表 |
| 最长递增子序列 | 最小化节点移动 |
