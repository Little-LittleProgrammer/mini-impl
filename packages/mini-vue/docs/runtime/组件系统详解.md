# 组件系统详解

## 什么是组件？

组件是 Vue 应用的基本构建块。每个组件包含自己的状态、模板和逻辑，可以复用和组合。

```typescript
// 组件定义
const MyComponent = {
  props: {
    title: String
  },
  setup(props) {
    const count = ref(0)
    return { count }
  },
  template: `
    <div>
      <h1>{{ title }}</h1>
      <button @click="count++">{{ count }}</button>
    </div>
  `
}
```

## 组件实例

### ComponentInternalInstance

```typescript
// packages/runtime-core/src/component.ts

export interface ComponentInternalInstance {
  // 唯一标识
  uid: number

  // VNode
  vnode: VNode

  // 组件选项
  type: ComponentOptions

  // 父组件实例
  parent: ComponentInternalInstance | null

  // 根组件实例
  root: ComponentInternalInstance

  // 渲染代理
  proxy: ComponentPublicInstance | null

  // 暴露给模板的属性
  exposed: Record<string, any> | null

  // setup 返回的状态
  setupState: Record<string, any>

  // props
  props: Record<string, any>
  attrs: Record<string, any>

  // 插槽
  slots: InternalSlots

  // emit
  emit: EmitFunction

  // 渲染上下文
  render: RenderFunction | null

  // 渲染后的子树
  subTree: VNode

  // effect 实例
  effect: ReactiveEffect

  // 更新函数
  update: SchedulerJob

  // 挂载状态
  isMounted: boolean

  // 生命周期钩子
  bc: LifecycleHook[] | null  // beforeCreate
  c: LifecycleHook[] | null   // created
  bm: LifecycleHook[] | null  // beforeMount
  m: LifecycleHook[] | null   // mounted
  bu: LifecycleHook[] | null  // beforeUpdate
  u: LifecycleHook[] | null   // updated
  bum: LifecycleHook[] | null // beforeUnmount
  um: LifecycleHook[] | null  // unmounted

  // 其他属性...
}
```

### 创建组件实例

```typescript
export function createComponentInstance(
  vnode: VNode,
  parent: ComponentInternalInstance | null
): ComponentInternalInstance {
  const instance: ComponentInternalInstance = {
    uid: uid++,
    vnode,
    type: vnode.type as ComponentOptions,
    parent,
    root: parent ? parent.root : instance,
    proxy: null,
    exposed: null,
    setupState: {},
    props: {},
    attrs: {},
    slots: {},
    emit: null as any,
    render: null,
    subTree: null!,
    effect: null!,
    update: null!,
    isMounted: false,
    bc: null,
    c: null,
    bm: null,
    m: null,
    bu: null,
    u: null,
    bum: null,
    um: null
  }

  // 创建 emit 函数
  instance.emit = emit.bind(null, instance)

  return instance
}
```

## setup 组件

```typescript
export function setupComponent(instance: ComponentInternalInstance) {
  const { props, children } = instance.vnode

  // 1. 初始化 props
  initProps(instance, props)

  // 2. 初始化 slots
  initSlots(instance, children)

  // 3. 执行 setup
  setupStatefulComponent(instance)
}
```

### setupStatefulComponent

```typescript
function setupStatefulComponent(instance: ComponentInternalInstance) {
  const Component = instance.type

  // 创建渲染代理
  instance.proxy = new Proxy(instance, instanceProxyHandler)

  // 获取 setup 函数
  const { setup } = Component

  if (setup) {
    // 设置当前实例
    setCurrentInstance(instance)

    // 执行 setup
    const setupResult = setup(
      instance.props,
      {
        emit: instance.emit,
        slots: instance.slots,
        attrs: instance.attrs,
        expose: (exposed) => { instance.exposed = exposed }
      }
    )

    // 重置当前实例
    unsetCurrentInstance()

    // 处理 setup 返回值
    handleSetupResult(instance, setupResult)
  } else {
    // 没有 setup，直接完成
    finishComponentSetup(instance)
  }
}
```

### handleSetupResult

```typescript
function handleSetupResult(
  instance: ComponentInternalInstance,
  setupResult: unknown
) {
  if (isFunction(setupResult)) {
    // setup 返回渲染函数
    instance.render = setupResult as RenderFunction
  } else if (isObject(setupResult)) {
    // setup 返回状态对象
    instance.setupState = proxyRefs(setupResult)
  }

  finishComponentSetup(instance)
}
```

### finishComponentSetup

```typescript
function finishComponentSetup(instance: ComponentInternalInstance) {
  const Component = instance.type

  // 如果没有 render 函数，尝试从模板编译
  if (!instance.render) {
    if (Component.template && !Component.render) {
      // 编译模板
      const { render } = compile(Component.template)
      Component.render = render
    }

    instance.render = Component.render!
  }

  // 兼容 Vue 2 选项式 API
  // ...
}
```

## 渲染代理

### instanceProxyHandler

```typescript
const instanceProxyHandler: ProxyHandler<ComponentInternalInstance> = {
  get({ type, setupState, props }, key) {
    // 1. setup 状态
    if (key in setupState) {
      return setupState[key]
    }

    // 2. props
    if (key in props) {
      return props[key]
    }

    // 3. 组件选项
    if (key in type) {
      return type[key]
    }

    // 4. 全局属性
    // ...
  },

  set({ setupState, props }, key, value) {
    if (key in setupState) {
      setupState[key] = value
      return true
    } else if (key in props) {
      // 警告：不能直接修改 props
      return false
    }
    return true
  }
}
```

## 处理组件

### processComponent

```typescript
const processComponent = (
  n1: VNode | null,
  n2: VNode,
  container: Element,
  anchor: Node | null
) => {
  if (n1 == null) {
    // 挂载组件
    mountComponent(n2, container, anchor)
  } else {
    // 更新组件
    updateComponent(n1, n2)
  }
}
```

### mountComponent

```typescript
const mountComponent = (
  vnode: VNode,
  container: Element,
  anchor: Node | null
) => {
  // 1. 创建组件实例
  const instance = createComponentInstance(vnode, null)

  // 保存实例引用
  vnode.component = instance

  // 2. setup 组件
  setupComponent(instance)

  // 3. 创建响应式 effect
  setupRenderEffect(instance, vnode, container, anchor)
}
```

### setupRenderEffect

```typescript
const setupRenderEffect = (
  instance: ComponentInternalInstance,
  vnode: VNode,
  container: Element,
  anchor: Node | null
) => {
  // 创建更新函数
  const componentUpdateFn = () => {
    if (!instance.isMounted) {
      // 挂载
      const { bm, m } = instance

      // beforeMount 钩子
      if (bm) {
        invokeArrayFns(bm)
      }

      // 渲染子树
      const subTree = (instance.subTree = renderComponentRoot(instance))

      // patch 子树
      patch(null, subTree, container, anchor)

      // 保存 DOM 引用
      vnode.el = subTree.el

      instance.isMounted = true

      // mounted 钩子
      if (m) {
        queuePostFlushCb(() => {
          invokeArrayFns(m)
        })
      }
    } else {
      // 更新
      const { bu, u } = instance
      const prevTree = instance.subTree

      // beforeUpdate 钩子
      if (bu) {
        invokeArrayFns(bu)
      }

      // 重新渲染
      const nextTree = renderComponentRoot(instance)
      instance.subTree = nextTree

      // patch 子树
      patch(prevTree, nextTree, container, anchor)

      vnode.el = nextTree.el

      // updated 钩子
      if (u) {
        queuePostFlushCb(() => {
          invokeArrayFns(u)
        })
      }
    }
  }

  // 创建 effect
  const effect = new ReactiveEffect(componentUpdateFn, () => queueJob(update))

  const update = instance.update = effect.run.bind(effect)

  // 首次执行
  update()
}
```

### renderComponentRoot

```typescript
function renderComponentRoot(instance: ComponentInternalInstance): VNode {
  const { render, setupState, props, proxy } = instance

  // 设置当前实例
  setCurrentInstance(instance)

  try {
    // 执行渲染函数
    const subTree = render!.call(proxy, proxy)
    return subTree
  } finally {
    unsetCurrentInstance()
  }
}
```

## 更新组件

### updateComponent

```typescript
const updateComponent = (n1: VNode, n2: VNode) => {
  const instance = (n2.component = n1.component)!

  // 检查是否需要更新
  if (shouldUpdateComponent(n1, n2)) {
    // 需要更新
    instance.next = n2
    instance.update()
  } else {
    // 不需要更新，直接复用
    n2.el = n1.el
    n2.vnode = n2
  }
}
```

### shouldUpdateComponent

```typescript
function shouldUpdateComponent(prevVNode: VNode, nextVNode: VNode): boolean {
  const { props: prevProps } = prevVNode
  const { props: nextProps } = nextVNode

  if (!nextProps) {
    return false
  }

  if (!prevProps) {
    return true
  }

  // 比较 props
  for (const key in nextProps) {
    if (nextProps[key] !== prevProps[key]) {
      return true
    }
  }

  return false
}
```

## Props 初始化

```typescript
export function initProps(
  instance: ComponentInternalInstance,
  rawProps: VNodeProps | null
) {
  const props: Record<string, any> = {}
  const attrs: Record<string, any> = {}

  const options = instance.type.props

  if (rawProps) {
    for (const key in rawProps) {
      const value = rawProps[key]
      if (options && key in options) {
        // 声明的 prop
        props[key] = value
      } else {
        // 未声明的，作为 attrs
        attrs[key] = value
      }
    }
  }

  instance.props = reactive(props)
  instance.attrs = attrs
}
```

## Slots 初始化

```typescript
export function initSlots(
  instance: ComponentInternalInstance,
  children: VNodeChildren
) {
  if (children && (children as any).__) {
    // 是插槽对象
    instance.slots = children as InternalSlots
  } else {
    instance.slots = {}
  }
}
```

## emit 实现

```typescript
export function emit(
  instance: ComponentInternalInstance,
  event: string,
  ...args: any[]
) {
  const props = instance.vnode.props || {}
  const handlerName = toHandlerKey(event)
  const handler = props[handlerName]

  if (handler) {
    // 调用事件处理器
    callWithAsyncErrorHandling(
      handler,
      instance,
      args
    )
  }
}

// 事件名转换
// click -> onClick
// update:modelValue -> onUpdate:modelValue
function toHandlerKey(event: string): string {
  return event ? `on${capitalize(event)}` : ``
}
```
